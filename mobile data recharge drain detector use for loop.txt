# include<stdio.h>
int main(){
    int totalData,n,i,dailyUsage,successfulDays=0;
    scanf("%d %d",&totalData,&n);
    for(i=0;i<n;i++){
        scanf("%d",&dailyUsage);
        if(totalData>=dailyUsage){
            totalData-=dailyUsage;
            successfulDays++;
        }
        else{
            break;
        }
    }
    printf("Remaining Data:%d\n",totalData);
    printf("Successful Days:%d",successfulDays);
    return 0;
}
