# include<stdio.h>
int main(){
    int n,i=0,delay,totalDelay=0,delayedDays=0;
    scanf("%d",&n);
    while(i<n){
        scanf("%d",&delay);
        totalDelay+=delay;
        if(delay>2){
            delayedDays++;
        }
        i++;
    }
    printf("Total Delay:%d\n",totalDelay);
    if(delayedDays>0){
        printf("Delayed Days:%d",delayedDays);
    }
    return 0;
}
