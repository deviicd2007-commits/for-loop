#include <stdio.h>

int main() {
    int n = 4;   
    int i, j;
    char ch;

    for(i = 1; i <= n; i++) {

        for(j = 1; j <= n - i; j++) {
            printf(" ");
        }

        
        ch = 'a';
        for(j = 1; j <= i; j++) {
            printf("%c", ch);
            ch++;
        }

       
        ch = ch - 2;   
        for(j = 1; j < i; j++) {
            printf("%c", ch);
            ch--;
        }

        printf("\n");
    }

    return 0;
}
