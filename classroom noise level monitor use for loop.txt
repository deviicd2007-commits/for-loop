# include<stdio.h>
int main(){
    int n,i,noise,maxnoise,noisecount=0;
    scanf("%d",&n);
    for(i=0;i<n;i++){
        scanf("%d",&noise);
        if(i==0){
            maxnoise=noise;
        }
        if(noise>maxnoise){
            maxnoise=noise;
        }
        if(noise>70){
            noisecount++;
        }
    }
    printf("Maximum Noise:%d\n",maxnoise);
    printf("Noise periods:%d",noisecount);
    return 0;
}
