#include <stdio.h>

int main() {
    int dataPack, N;
    int usage;
    int i = 0;
    int exhaustedDay = -1;
    int overused = 0;

    scanf("%d", &dataPack);
    scanf("%d", &N);

    while (i < N) {
        scanf("%d", &usage);
        dataPack -= usage;

        if (dataPack <= 0 && exhaustedDay == -1) {
            exhaustedDay = i + 1;          
            overused = -dataPack;          
            break;                      
        }

        i++;
    }

    if (exhaustedDay == -1) {
        printf("Exhausted Day: Not Exhausted\n");
        printf("Overused Data: 0");
    } else {
        printf("Exhausted Day: %d\n", exhaustedDay);
        printf("Overused Data: %d", overused);
    }

    return 0;
}
