# include<stdio.h>
int main(){
    int n,i;
    scanf("%d",&n);
    int sales[n];
    for(i=0;i<n;i++){
        scanf("%d",&sales[i]);
    }
    int isSorted=1;
    for(i=0;i<n-1;i++){
        if(sales[i]>sales[i+1]){
            isSorted=0;
        }
    }
    if(isSorted){
        printf("yes");
    }else{
        printf("no");
    }
    return 0;
}
