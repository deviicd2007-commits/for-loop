# include<stdio.h>
int main(){
    int n,i,expense,total=0,count=0;
    scanf("%d ",&n);
    for(i=0;i<n;i++){
        scanf("%d",&expense);
        total +=expense;
        if(expense> 1000){
            count++;
        }
    }
    printf("Total Expense :%d\n",total);
    printf("Overspend Days:%d",count);
    return 0;
}
