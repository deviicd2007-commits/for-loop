# include<stdio.h>
int main(){
    int n,i,total=0,heavyDays=0,rainfall;
    scanf("%d",&n);
    for(i=0;i<n;i++){
        scanf("%d",&rainfall);
       total+=rainfall ;
       if(rainfall>50){
           heavyDays++;
       }
    }
    printf("Total Rainfall:%d\n",total);
    printf("Heavy Rain Days:%d",heavyDays);
    return 0;
}
