 #include <stdio.h>

int main() {
    int n, i, j, count = 0;

    scanf("%d", &n);

    for (i = 2; i <= n; i++) {
        int num = 1;   

        for (j = 2; j < i; j++) {   
            if (i % j == 0) {
                num = 0;
                break;
            }
        }

        if (num == 1) {
            count++;
        }
    }

    printf("Prime number count: %d", count);

    return 0;
}
